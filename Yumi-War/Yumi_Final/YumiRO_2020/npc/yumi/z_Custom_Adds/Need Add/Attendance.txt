    if (#lastDayOnline < gettimetick(2)/86400) {

            set #consecutive_days, #consecutive_days+1;

         set #lastDayOnline, gettimetick(2)/86400;

            set #CASHPOINTS, #KAFRAPOINTS + 200;        // Jellopy

    end;
}
